#include<stdio.h>
//#include<conio.h>
int main()
{
int n,i,flag=0;
//clrscr();
printf("To Check if the Number is Prime \n");
printf("Enter a positive integer:");
scanf("%d",&n);
for(i=2;i<n/2;i++)
{
	if(n%i==0)
	{
	   flag=1;
	   break;
	}
}
if(flag==0)

printf("\n %d is a prime number",n);

else

printf("\n %d is not a prime number",n);

}
