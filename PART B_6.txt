#include<stdio.h> 
//#include<conio.h>

int gcd(int m, int n)
{
if (n==0)
return(m);
else if(n>m) 
return(gcd(n,m));
else
return(gcd(n,m%n));
}

int main()
{
int m,n;
//clrscr();
printf ("enter m and n \n");
scanf("%d %d",&m,&n);
printf ("GCD (%d,%d) = %d\n",m,n,gcd(m,n));
//getch();
}
